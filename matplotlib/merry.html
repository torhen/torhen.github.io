import matplotlib.pyplot as plt
from matplotlib.patches import Polygon, Circle
from matplotlib.animation import FuncAnimation, PillowWriter
from matplotlib import patheffects

plt.xkcd()
plt.rcParams['path.effects'] = [patheffects.withStroke(linewidth=0)]

vertices = [
    (-1, 0),
    (-1, 1),
    (-3, 1),
    (-1, 3),
    (-3, 3),
    (-1, 5),
    (-2, 5),
    (0, 7),
    (2, 5),
    (1, 5),
    (3, 3),
    (1, 3),
    (3, 1),
    (1, 1),
    (1, 0),
]

polygon = Polygon(vertices, facecolor='green', edgecolor='black')

fig = plt.figure(figsize=(2,2), facecolor='blue')
ax = fig.add_subplot(111)
ax.set_axis_off()
ax.set_xlim(-4, 4)
ax.set_ylim(0, 8)
ax.add_patch(polygon)
ax.text(x=0, y=8, s="Merry Christmas!", fontsize=13, color='white', ha='center')


cir1 = Circle([-0.5,2], 0.2, facecolor='red', edgecolor='black')
ax.add_patch(cir1)

cir2 = Circle([0.5,3.5], 0.2, facecolor='blue', edgecolor='black')
ax.add_patch(cir2)

cir3 = Circle([-0.5,5], 0.2, facecolor='blue', edgecolor='black')
ax.add_patch(cir3)


def animation(frame):
    if frame % 2 == 0:
        cir1.set_facecolor('red')
        cir2.set_facecolor('cyan')
        cir3.set_facecolor('yellow')

    else:
        cir1.set_facecolor('cyan')
        cir2.set_facecolor('yellow')
        cir3.set_facecolor('red')


ani = FuncAnimation(fig, animation, interval=500)

# Save animation as a GIF
writer = PillowWriter(fps=2)  # Adjust FPS as needed
ani.save('merry.gif', writer=writer)

plt.show()
